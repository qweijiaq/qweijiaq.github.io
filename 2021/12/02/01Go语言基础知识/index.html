变量和常量

1. 定义变量

package main

import "fmt"

// 可以同时定义多个变量
// 全局变量不能使用短变量声明
//全局变量可以声明后不使用
var (
	aa =3
	ss = "kkk"
	bb = true
)

// 定义未赋初值的变量
func variableZeroValue() {
	// go语言定义变量时，变量名在前，变量类型在后
	var a int    // 不同于其他语言未初始化的变量为随机值，go语言未初始化的int类型变量初始值为0
	var s string // go语言未初始化的string类型变量初始值为空字符串
	fmt.Printf("%d %q\n", a, s)
}

// 定义赋初值的变量
func variableInitialValue() {
	// go语言中局部变量一旦定义则必须使用，否则编译无法通过
	var a, b int = 3, 4
	var s string = "abc"
	fmt.Println(a, b, s)
}

// 定义变量时可以省略变量类型，而通过初始值的类型来自动判断变量类型
func variableTypeDeduction() {
	var a, b, c, s = 3, 4, true, "def"  // 平行赋值多个变量
	fmt.Println(a, b, c, s)
}

// 短变量声明
func variableShorter() {
	// 这种短变量声明只能用在函数内部，且只能用于变量第一次声明
	// 这也是最推荐的一种写法
	a, b, c, s := 3, 4, true, "def"
	b = 5
	fmt.Println(a, b, c, s)
}

func main() {
	fmt.Println("Hello World")
	variableZeroValue()
	variableInitialValue()
	variableTypeDeduction()
	variableShorter()
}

运行结果：

Hello World
0 ""
3 4 abc
3 4 true def
3 5 true def

总结：

  1. 使用var关键字
    • var a, b, c bool // 默认均为false
    • ``var s1, s2 string = “hello world”`
    • 可以放在函数内部（局部），或直接放在包内（全局）
    • 可以使用var()集中定义变量
  2. 让编译器自动决定类型
    • var a, b, i, s1, s2 = true, false, 3, "hello", "world"
  3. 使用:=定义变量，这是推荐的写法
    • a, b, i, s1, s2 := true, false, 3, "hello", "world"
    • 只能在函数内部使用

2. 内建变量类型

package main

import (
	"fmt"
	"math"
	"math/cmplx"
)

// 欧拉公式 e ^ {i * pi} + 1 = 0
func euler() {
	c := 3 + 4i
	// cmplx.Abs() -- 对复数取模
	fmt.Println(cmplx.Abs(c))
	// 理论上答案是0，但是由于浮点数存在误差，故结果可能存在很小的误差
	fmt.Println(cmplx.Exp(1i*math.Pi) + 1)
	fmt.Println(cmplx.Pow(math.E, 1i*math.Pi) + 1)
	fmt.Printf("%.3f\n", cmplx.Exp(1i*math.Pi)+1) // 取小数点后三位
}

// 勾股定理
func triangle() {
	var a, b int = 3, 4
	var c int
	// math.Sqrt()的参数变量是一个float64类型,返回值也是一个float64类型
    // Go语言只有强制类型转换，没有隐式转换
	// 强制类型转换 int() float64() ...
	c = int(math.Sqrt(float64(a*a + b*b)))
	fmt.Println(c)
}

func main() {
	euler()
	triangle()
}

运行结果：

5
(0+1.2246467991473515e-16i)
(0+1.2246467991473515e-16i)
(0.000+0.000i)
5

内建变量类型：

  • bool,string
  • (u)int, (u)int8, (u)int16, (u)int32, (u)int64, uintptr
  • byte（8位，ASCII码，只支持英文字符）, rune（32位，Unicode，支持中文和其他语言）
    • go语言没有char类型,只有rune类型
  • float32, float64, complex64, complex128
  • Go语言原生支持复数

3. 常量和枚举

package main

import (
	"fmt"
	"math"
)

func consts() {
	// 指定常量类型
	const filename string = "abc.txt"
	// 不指定常量类型，既可以作为int类型，也可以是浮点型，取决于它作用时的需求
	const a, b = 3, 4
	var c int
	// 这时我们发现不需要将a*a+b*b转换为float64类型，因为系统可以自动将其认定为float64型
	c = int(math.Sqrt(a*a + b*b))
	// 在下面的环境中，a和b又被认为是int型
	fmt.Printf("%s %d %T %T\n", filename, c, a, b)
	fmt.Println("----------")
}

// 普通枚举
func enums0() {
	const (
		cpp    = 0
		java   = 1
		python = 2
		golang = 3
	)
	fmt.Println(cpp, java, python, golang)
	fmt.Println("----------")
}

// 自增枚举，每出现一次const关键字，iota自动置0，每定义一个常量，iota自动加1
func enums1() {
	const (
		cpp = iota
		java
		python
		golang
	)
	fmt.Println(cpp, java, python, golang)
	fmt.Println("----------")
}

func enums2() {
	const (
		cpp = iota
		_
		python
		golang
		javascript
	)

	// b, kb, mb, gb, tb, pb
	const (
		b = 1 << (10 * iota)
		kb
		mb
		gb
		tb
		pb
	)
	fmt.Println(cpp, javascript, python, golang)
	fmt.Println("----------")
	fmt.Println(b, kb, mb, gb, tb, pb)
}

func main() {
	consts()
	enums0()
	enums1()
	enums2()
}

1. 条件语句

package main

import (
	"fmt"
	"io/ioutil"
)

func eval(a, b int, op string) int {
	var result int
	switch op {
	case "+":
		result = a + b
	case "-":
		result = a - b
	case "*":
		result = a * b
	case "/":
		result = a / b
	default:
		panic("unsupported operator:" + op)
	}
	return result
}

func grade(score int) string {
	g := ""
	switch {  // switch后面可以没有表达式
	case score < 0 || score > 100:
		panic(fmt.Sprintf("Wrong score: %d", score))
	case score < 60:
		g = "F"
	case score < 80:
		g = "C"
	case score < 90:
		g = "B"
	case score <= 100:
		g = "A"
	}
	return g
}

func main() {
	const filename = "abc.txt"
	contents, err := ioutil.ReadFile(filename) // ReadFile返回的第一个值为文件内容，第二个值为error值，方便判断是否报错，没有问题时返回nil，报错时返回错误信息
	if err != nil {
		fmt.Println(err)
	} else {
		fmt.Printf("%s\n", contents)
	}
	/* Go语言中还可采用如下写法：
	if contents,err := ioutil.ReadFile(filename); err != nil {
		fmt.Println(err)
	} else {
		fmt.Printf("%s\n", contents)
	}
	fmt.Printf("%s\n", contents)  // 因为contents定义在if语句中，所以contents变量不能在if语句范围外使用，故此行代码会报错
	*/

	fmt.Println(
		grade(0),
		grade(59),
		grade(60),
		grade(82),
		grade(99),
		grade(100),
		grade(101), // 因为括号换行了，所以结尾要加一个逗号
	)
}

总结：

  • if语句
    • if的条件里不需要括号
    • if的条件里可以赋值
    • if的条件里赋值的变量作用域就在这个if语句里
  • switch语句
    • switch语句不需要break，会自动break，除非使用fallthrough
    • switch后面可以没有表达式
    • switch也可以直接跟多个条件

2. 循环语句

package main

import (
	"bufio"
	"fmt"
	"os"
	"strconv"
)

// 将十进制数转换为二进制数
func convertToBin(n int) string {
	result := ""
	for ; n > 0; n /= 2 {
		lsb := n % 2
		result = strconv.Itoa(lsb) + result // strconv.Itoa()将int型转换为字符串型
	}
	return result
}

// 打印文本内容
func printFile(filename string) {
	file, err := os.Open(filename)
	if err != nil {
		panic(err)
	}

	scanner := bufio.NewScanner(file)

	// scanner.Scan()为结束条件，此时可以不写分号
	for scanner.Scan() {
		fmt.Println(scanner.Text())
	}
}

// 死循环
func forever() {
	for {
		fmt.Println("abc")
	}
}

func main() {
	fmt.Println(
		convertToBin(5),  // 101 --> 101  一直对2取余后倒置
		convertToBin(13), // 1011 --> 1101
		convertToBin(238417),
		convertToBin(0), // 空串不显示出来
	)

	printFile("abc.txt")
}

运行结果：

101 1101 111010001101010001 
abcdef
123456

总结：

  • for的条件里不需要括号
  • for的条件里可以省略初始条件、结束条件、递增表达式
    * 省略初始条件相当于`while`
* 省略三项就是一个死循环
  • Go语言中没有while循环

函数

package main

import (
	"fmt"
	"math"
	"reflect"
	"runtime"
)

// 返回一个值
func eval(a, b int, op string) int {
	var result int
	switch op {
	case "+":
		result = a + b
	case "-":
		result = a - b
	case "*":
		result = a * b
	case "/":
		result = a / b
	default:
		panic("unsupported operator:" + op)
	}
	return result
}

// 返回多个值
func div1(a, b int) (int, int) {
	return a / b, a % b
}

func div2(a, b int) (q, r int) {
	q = a / b
	r = a % b
	return // 给返回值命名后，return时可以不写变量名
}

// 两个返回值常用的场景：第二个返回值返回错误信息
func eval2(a, b int, op string) (int, error) {
	switch op {
	case "+":
		return a + b, nil
	case "-":
		return a + b, nil
	case "*":
		return a + b, nil
	case "/":
		return a + b, nil
	default:
		return 0, fmt.Errorf("unsupported operator:" + op)
	}
}

// 函数式编程
// apply函数的第一个参数为op，op是一个函数，函数的参数是两个int类型，且返回值是一个int类型
// apply函数的第2和3个参数为a和b，均为int类型
// apply函数的返回值是一个int类型
func apply(op func(int, int) int, a, b int) int {
	p := reflect.ValueOf(op).Pointer()    // 获取op函数的指针
	opName := runtime.FuncForPC(p).Name() // 获取op函数的函数名
	fmt.Printf("Calling function %s with args"+"(%d, %d)\n", opName, a, b)
	return op(a, b)
}

func pow(a, b int) int {
	return int(math.Pow(float64(a), float64(b)))
}

// ...表示可以传任意个参数
func sum(numbers ...int) int {
	s := 0
	for i := range numbers {
		s += numbers[i]
	}
	return s
}

func main() {
	fmt.Println(eval(3, 4, "*"))

	fmt.Println(div1(13, 3))

	q, r := div2(14, 3)
	_, s := div2(15, 3) // 不想接收的值可以使用占位符(匿名变量)替代
	fmt.Println(q, r, s)

	fmt.Println(eval2(3, 4, "x"))

	fmt.Println(apply(pow, 3, 4))

	// 直接在apply函数内部定义op函数
	fmt.Println(apply(
		func(a int, b int) int {
			return int(math.Pow(float64(a), float64(b)))
		}, 3, 5))

	fmt.Println(sum(1, 2, 3, 4, 5))
}

运行结果：

12
4 1
4 2 0
0 unsupported operator:x
Calling function main.pow with args(3, 4)  // 第一个mian表示main包名，第二个pow表示pow函数
81
Calling function main.main.func1 with args(3, 5)  // 第一个main表示main包名，第二个main表示main函数名，第三个func1表示未知的函数名
243
15

总结：

  1. 函数可以返回多个值
  2. 函数返回多个值时可以起名字，这仅用于非常简单的函数，且对于调用者而言没有区别，仅仅起提示作用
  3. 函数中可以传入可变参数列表
  4. 返回值类型写在最后面
  5. 函数可以作为参数
  6. 没有默认参数、可选参数的概念

指针

package main

import "fmt"

// 值传递，不改变原来的值
func swap1(a, b int) {
	b, a = a, b
}

// 地址传递
func swap2(a, b *int) {
	*b, *a = *a, *b
}

// 直接通过return实现交换
func swap3(a, b int) (int, int) {
	return b, a
}

func main() {
	a, b := 3, 4
	swap1(a, b)
	fmt.Println(a, b)
	swap2(&a, &b)
	fmt.Println(a, b)
	c, d := 1, 2
	c, d = swap3(c, d)
	fmt.Println(c, d)
}

运行结果：

3 4
4 3
2 1

总结：

  • Go语言中的指针不能运算
  • Go语言只有值传递一种方式，不过可以通过指针实现地址传递

数组

package main

import "fmt"

// 打印数组
// 参数可以是arr1和arr3，不可以是arr2
func printArray1(arr [5]int) {
	for i, v := range arr {
		fmt.Println(i, v)
	}
}

// 数组是值传递，在函数内部改变数组的元素只会影响函数内部，函数外部值不变
func printArray2(arr [5]int) {
	arr[0] = 100
	for i, v := range arr {
		fmt.Println(i, v)
	}
}

// 采用地址传递
func printArray3(arr *[5]int) {
	arr[0] = 100
	for i, v := range arr {
		fmt.Println(i, v)
	}
}

func main() {
	// 定义一个未初始化的数组，包含5个元素，int类型数组未初始化时默认元素全为0
	var arr1 [5]int
	// 定义一个初始化的数组，包含3个元素
	arr2 := [3]int{1, 3, 5}
	//  定义一个初始化的数组，由编译器自己获取元素个数
	arr3 := [...]int{2, 4, 6, 8, 10}
	// 定义二维数组
	var grid [4][5]int

	fmt.Println(arr1, arr2, arr3)
	fmt.Println(grid)

	// 遍历数组
	// i为索引，v为值
	for i, v := range arr3 {
		fmt.Println(i, v)
	}

	printArray1(arr1)
	printArray1(arr3)
	// printArray(arr2)是错误的，因为函数要求参数为5个元素的数组，而arr2只有三个元素，Go语言认为是不同的数据类型

	fmt.Println("采用值传递")
	fmt.Println("printArray2(arr1)")
	printArray2(arr1)
	fmt.Println("printArray2(arr3)")
	printArray2(arr3)
	fmt.Println("print arr1 and arr3")
	fmt.Println(arr1, arr3)

	fmt.Println("采用地址传递")
	fmt.Println("printArray3(arr1)")
	printArray3(&arr1)
	fmt.Println("printArray3(arr3)")
	printArray3(&arr3)
	fmt.Println("print arr1 and arr3")
	fmt.Println(arr1, arr3)
}

/*

	4. 为什么要用range
		* 意义明确，美观
		* c++：原生没有类似能力
		* Java/Python：只能for each value，不能同时获取i，v
	5. [10]int和[20]int是不同类型
	6. 调用func f(arr [10]int)会拷贝数组
	7. 在go语言中一般不直接使用数组
*/

切片

package main

import "fmt"

// slice是array的一个视图，可通过修改array的slice得到array地址传递的效果
func updateSlice(s []int) {
	s[0] = 100
}

func printSlice(s []int) {
	fmt.Printf("%v, len=%d, cap=%d\n", s, len(s), cap(s))
}

func main() {
	arr := [...]int{0, 1, 2, 3, 4, 5, 6, 7}
	fmt.Println("arr[2:6]", arr[2:6]) // 2, 3, 4, 5
	fmt.Println("arr[:6]", arr[:6])   // 0, 1, 2, 3, 4, 5
	fmt.Println("arr[2:]", arr[2:])   // 2, 3, 4, 5, 6, 7
	fmt.Println("arr[:]", arr[:])     // 0, 1, 2, 3, 4, 5, 6, 7

	s1 := arr[2:]
	fmt.Println("before s1: ", s1)
	fmt.Println("After updateSlice(s1)")
	updateSlice(s1)
	fmt.Println("after s1: ", s1)
	fmt.Println("after arr: ", arr)

	fmt.Println("Reslice")
	fmt.Println(s1)
	s1 = s1[:5]
	fmt.Println(s1)
	s1 = s1[2:]
	fmt.Println(s1)

	// slice的扩展
	fmt.Println("arr: ", arr)
	s1 = arr[2:6]
	fmt.Println("s1 = arr[2:6]", s1)
	s2 := s1[3:5]
	fmt.Println("s2 = s1[3:5]", s2)

	fmt.Printf("s1=%v, len(s)=%d, cap (s)=%d\n", s1, len(s1), cap(s1))

	// 向slice添加元素
	arr1 := [...]int{0, 1, 2, 3, 4, 5, 6, 7}
	a1 := arr1[2:6]
	a2 := a1[3:5]
	a3 := append(a2, 10)
	a4 := append(a3, 11)
	a5 := append(a4, 12)
	fmt.Println(arr1)
	// a4 and a5 no longer view arr1
	fmt.Println(a3, a4, a5)
	fmt.Println(arr1)

	// 直接定义一个slice
	var s []int // Zero value for slice is nil   s == nil

	for i := 0; i < 10; i++ {
		printSlice(s)
		s = append(s, 2*i+1)
	}
	fmt.Println(s)

	// 创建slice并初始化
	ss1 := []int{2, 4, 6, 8}
	// 使用make函数创建slice
	// 接收两个参数时,第二个参数为len和cap的值
	// 接收三个参数时,第二个参数为len,第三个参数为cap
	ss2 := make([]int, 17)
	ss3 := make([]int, 10, 32)
	printSlice(ss1)
	printSlice(ss2)
	printSlice(ss3)

	// 切片拷贝
	fmt.Println("Copying slice")
	copy(ss2, ss1)
	printSlice(ss2)

	// 删除切片元素,没有内置方法,可以使用切片+append实现
	fmt.Println("Deleting elements from slice")
	ss2 = append(ss2[:3], ss2[4:]...)
	printSlice(ss2)

	fmt.Println("Popping from front")
	front := ss2[0]
	ss2 = ss2[1:]
	fmt.Println(front)
	printSlice(ss2)

	fmt.Println("Popping from back")
	tail := ss2[len(ss2)-1]
	ss2 = ss2[:len(ss2)-1]

package main

func main() {
	// 直接创建map
	m := map[string]string{
		"name":    "ccmouse",
		"course":  "golang",
		"site":    "imooc",
		"quality": "notbad",
	}
	// 使用make创建map
	m2 := make(map[string]string) // m2 == empty map
	// 创建为nil的map
	var m3 map[string]int // m3 == nil
	fmt.Println(m, m2, m3)

	// map的遍历
	for k, v := range m {
		fmt.Println(k, v)
	}

	// Getting values
	fmt.Println("Getting values")
	courseName := m["course"]
	fmt.Println(courseName)
	causeName := m["cause"]
	fmt.Println(causeName) // 从map通过不存在的key获取value时不会报错，返回一个空值
	// 可以通过多接收一个变量来判断key是否存在于map里
	causeName, ok := m["cause"]
	fmt.Println(causeName, ok)
	courseName, ok = m["course"]
	fmt.Println(courseName, ok)
	if causeName, ok := m["cause"]; ok {
		fmt.Println(causeName)
	} else {
		fmt.Println("key does not exist")
	}

	// delete elements
	delete(m, "name")
	fmt.Println(m)
}

寻找最长不含有重复字符的子串

  • abcabcbb --> abc
  • bbbbb --> b
  • pwwkew --> wke

分析：

对于每一个字母x

  • lastOccurred[x]不存在，或者< start，则无需操作
  • lastOccurred[x] >= start，则更新start
  • 更新lastOccurred[x]，更新maxLength
package main

import (
	"fmt"
	"unicode/utf8"
)

func lengthOfNonRepeatingSubStr(s string) int {
	lastOccurred := make(map[byte]int)
	start := 0
	maxLength := 0

	for i, ch := range []byte(s) {
		lastI, ok := lastOccurred[ch]
		if ok && lastI >= start {
			start = lastOccurred[ch] + 1
		}
		if i-start+1 > maxLength {
			maxLength = i - start + 1
		}
		lastOccurred[ch] = i
	}

	return maxLength
}

// 上面的函数不支持中文，要想支持中文，需要使用rune类型
func lengthOfNonRepeatingSubStr2(s string) int {
	lastOccurred := make(map[rune]int)
	start := 0
	maxLength := 0

	for i, ch := range []rune(s) {
		lastI, ok := lastOccurred[ch]
		if ok && lastI >= start {
			start = lastOccurred[ch] + 1
		}
		if i-start+1 > maxLength {
			maxLength = i - start + 1
		}
		lastOccurred[ch] = i
	}

	return maxLength
}

func main() {
	fmt.Println(lengthOfNonRepeatingSubStr("abcabcbb"))
	fmt.Println(lengthOfNonRepeatingSubStr("bbb"))
	fmt.Println(lengthOfNonRepeatingSubStr("pwwkew"))
	fmt.Println(lengthOfNonRepeatingSubStr(""))
	fmt.Println(lengthOfNonRepeatingSubStr("b"))
	fmt.Println(lengthOfNonRepeatingSubStr("abcdef"))

	s := "Yes我爱慕课网!"   // 每个汉字占3字节
	for _, b := range []byte(s) {
		fmt.Printf("%X ", b)
	}
	fmt.Println()

	for i, ch := range s { // ch is a rune
		fmt.Printf("(%d %X) ", i, ch)
	}
	fmt.Println()

	fmt.Println("Rune count: ",
		utf8.RuneCountInString(s))

	bytes := []byte(s)
	for len(bytes) > 0 {
		ch, size := utf8.DecodeRune(bytes)
		bytes = bytes[size:]
		fmt.Printf("%c ", ch)
	}
	fmt.Println()

	for i, ch := range []rune(s) {   // rune占四个字节
		fmt.Printf("(%d %c) ", i, ch)
	}
	fmt.Println()

	fmt.Println(lengthOfNonRepeatingSubStr2("一二三二一"))
	fmt.Println(lengthOfNonRepeatingSubStr2("这里是慕课网"))
}

package main

import "fmt"

type treeNode struct {
	value       int
	left, right *treeNode
}

// 自定义创建结构体的工厂函数
func createNode(value int) *treeNode {
	return &treeNode{value: value} // 在Go语言中可以返回局部变量的地址
}

func (node treeNode) Print() { // (node Node) Print()表明Print函数的接收者为node，即node.Print()
	fmt.Println(node.value)
}

// 值传递
func (node treeNode) SetValue1(value int) {
	node.value = value
}

// 地址传递
func (node *treeNode) SetValue2(value int) {
	if node == nil {
		fmt.Println("Setting value to nil node. Ignored.")
		return
	}
	node.value = value
}

// 树的遍历
// 左 - 中 - 右
func (node *treeNode) Traverse() {
	if node == nil {
		return
	}
	node.left.Traverse()  //左
	node.Print()          // 中
	node.right.Traverse() // 右
}

func main() {
	// 创建一个结构体，不进行初始化
    var root treeNode

	// 利用结构体创建树
	root = treeNode{value: 3}
	root.left = &treeNode{}
	root.right = &treeNode{5, nil, nil}
	root.right.left = new(treeNode)
	root.left.right = createNode(2)

	root.right.left.SetValue1(4)
	root.right.left.Print() // 值没有发生改变，说明是值传递
	root.right.left.SetValue2(4)
	root.right.left.Print() // 值发生改变，说明是地址传递

	root.Print()
	root.SetValue2(100)

	// Go语言中值传递和地址传递转换十分灵活
	PRoot := &root       // 如果函数或方法需要接收地址，则直接将pRoot传入；如果函数或方法需要接收值，则会自动将pRoot对应的值传入
	PRoot.Print()        // 传入pRoot对应的值
	PRoot.SetValue2(200) // 传入pRoot本身
	PRoot.Print()

	// 创建结构体切片
	nodes := []treeNode{
		{value: 3},
		{},
		{6, nil, &root},
	}
	fmt.Println(nodes)

	// nil指针也可以调用方法
	var pointRoot *treeNode  // nil指针
	pointRoot.SetValue2(200) // 调用了方法
	pointRoot = &root
	pointRoot.SetValue2(300)
	pointRoot.Print()

	root.Traverse()
}

		return

	f(node)

package main

import (
	"fmt"

	"github.com/imooc/tree"
)

type myTreeNode struct {
	node *tree.Node
}

// 通过别名
// 左 - 右 - 中
func (myNode *myTreeNode) postOrder() {
	if myNode == nil || myNode.node == nil {
		return
	}
	left := myTreeNode{myNode.node.Left}
	left.postOrder() // 左
	right := myTreeNode{myNode.node.Right}
	right.postOrder()   // 右
	myNode.node.Print() // 中
}

func main() {
	// 创建一个结构体，不进行初始化
	var root = tree.Node{Value: 3}
	root.Left = &tree.Node{}
	root.Right = &tree.Node{5, nil, nil}
	root.Right.Left = new(tree.Node)
	root.Left.Right = tree.CreateNode(2)
	root.Right.Left.SetValue(4)

	fmt.Print("In-order traversal: ")
	fmt.Println()
	root.Traverse()

	fmt.Printf("My-order traversal:\n")
	myRoot := myTreeNode{&root}
	myRoot.postOrder()
	fmt.Println()

	fmt.Printf("traverse by funcation:\n")
	root.TraverseFunc(func(n *tree.Node) {
		n.Print()
	})
	fmt.Println()

	nodeCount := 0
	root.TraverseFunc(func(node *tree.Node) {
		nodeCount++
	})
	fmt.Println("Node count:", nodeCount)
}

package queue

type Queue []interface{}

func (q *Queue) Push(v interface{}) {
	*q = append(*q, v)
}

func (q *Queue) Pop() interface{} {
	head := (*q)[0]
	*q = (*q)[1:]
	return head
}

func (q *Queue) IsEmpty() bool {
	return len(*q) == 0
}
